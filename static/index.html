Request Structure

POST /chat
Content-Type: application/json

{
  "message": "Your message here",
  "conversation_id": "optional-conversation-id",
  "agent_id": "optional-specific-agent-id"
}

Response Structure

{
  "response": "AI generated response",
  "agent_id": "agent_12345",
  "agent_name": "Marketing Assistant",
  "conversation_id": "conv_20241201_143022",
  "timestamp": "2024-12-01T14:30:22.123Z",
  "agent_reasoning": "Selected based on query analysis",
  "confidence": 0.95,
  "sources": [
    {
      "title": "Web Result",
      "url": "https://example.com"
    }
  ]
}

JavaScript Example

async function sendMessage(message, conversationId = null) {
  const response = await fetch('YOUR_INTEGRATION_URL/chat', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      message: message,
      conversation_id: conversationId
    })
  });
  
  const data = await response.json();
  return data;
}
